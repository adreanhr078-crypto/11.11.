const seed = localStorage.getItem("playerSeed") || Math.random().toString(36).slice(2);

function getVariant(text) {
  let hash = 0;
  for (let i = 0; i < seed.length; i++) {
    hash += seed.charCodeAt(i);
  }
  return hash % text.length;
}

  return new Date().toISOString().split("T")[0];
}

  const key = getDayKey();

  return `content_${key}`;
}

🧠 3) ذكاء اصطناعي يتطور عبر الأيام
let aiLevel = parseInt(localStorage.getItem("aiLevel") || "1");

function evolveDaily() {
  const lastVisit = localStorage.getItem("lastVisit");

  if (lastVisit !== getDayKey()) {
    aiLevel++;
    localStorage.setItem("aiLevel", aiLevel);
    localStorage.setItem("lastVisit", getDayKey());
  }
}
