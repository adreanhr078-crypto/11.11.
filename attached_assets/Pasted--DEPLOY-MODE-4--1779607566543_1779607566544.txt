🎮 3) تشغيل اللعبة (Game Entry)
function startGame() {
  document.getElementById("title").innerText = "WELCOME";

  speak("You are now inside.");

  audio.start();
  aiInit();
  startWorld();
}
🎧 4) تشغيل الصوت + AI معًا
function aiInit() {
  speak("I remember you...");
}

function startWorld() {
  setInterval(() => {
    showGlobalEvent();
    increaseCorruption();
  }, 10000);
}
