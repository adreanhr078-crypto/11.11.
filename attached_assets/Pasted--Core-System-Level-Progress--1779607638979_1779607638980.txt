1️⃣ حفظ التقدم (Local Storage)
function saveLevel(level) {
  localStorage.setItem("currentLevel", level);
}

function getLevel() {
  return parseInt(localStorage.getItem("currentLevel") || "1");
}

"أنت لست وحدك هنا..."

TH3 V01CE IS W4TCH1NG Y0U

THE VOICE IS WATCHING YOU
كود التحقق:
function checkLevel1(input) {
  if (input.toLowerCase() === "the voice is watching you") {
    saveLevel(2);
    return "LEVEL 2 UNLOCKED";
  }
  return "TRY AGAIN";
}

function checkTime() {
  const hour = new Date().getHours();

    return true;
  }
  return false;
}
