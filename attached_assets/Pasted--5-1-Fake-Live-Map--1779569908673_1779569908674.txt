🌍 1) خريطة لاعبين حيّة (Fake Live Map)
🧠 الفكرة

تعطي إحساس إن في لاعبين حول العالم داخل نفس التجربة.

function generateFakeMapPlayers() {
  const countries = ["Jordan", "Egypt", "USA", "UK", "Unknown"];
  
  let players = [];

  for (let i = 0; i < 12; i++) {
    players.push({
      name: "User_" + Math.floor(Math.random() * 9999),
      country: countries[Math.floor(Math.random() * countries.length)],
      status: Math.random() > 0.5 ? "ACTIVE" : "WATCHING"
    });
  }

  return players;
}
👁️ عرضهم:
function showMap() {
  const players = generateFakeMapPlayers();

  players.forEach(p => {
    console.log(`${p.name} from ${p.country} is ${p.status}`);
  });
}

🧬 2) رسائل من المستقبل (Future Messages)
🧠 الفكرة

الموقع يرسل “تنبؤات وهمية نفسية”

const futureMessages = [
  "Tomorrow you will remember this moment.",
  "You will return here at 3:11 AM.",
  "You already saw this message before.",
  "Do not trust the next choice."
];

function getFutureMessage() {
  return futureMessages[Math.floor(Math.random() * futureMessages.length)];
}
🔮 عرض الرسالة:
function showFuture() {

  speak(msg);
  audio.playGlitch();
}

🕳️ 5) النهاية المتغيرة حسب البلد (Location Ending)
🧠 الفكرة

النهاية تختلف حسب الدولة (وهمية داخل اللعبة فقط)

function getEndingByCountry(country) {
  switch (country) {
    case "Jordan":
      return "ENDING_DESERT_SIGNAL";

    case "USA":
      return "ENDING_CITY_NOISE";

    default:
      return "ENDING_UNKNOWN";
  }
}
🎬 تنفيذ النهاية:
function showGeoEnding() {
  const fakeCountry = generateFakeMapPlayers()[0].country;
  const ending = getEndingByCountry(fakeCountry);

  speak("Your location has been recorded...");

  console.log("ENDING:", ending);
}
