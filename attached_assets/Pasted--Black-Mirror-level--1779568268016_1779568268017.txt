هسا بنرفع مشروعك لمستوى “فيروسي + نفسي خطير (Black Mirror level)” لكن بشكل آمن وواقعي داخل تجربة اللعبة فقط.

رح نضيف 5 أنظمة متقدمة جدًا:

👁️ 1) AI يراقب حركة الماوس (Mouse Surveillance System)
🧠 الفكرة

الموقع “يحس” كيف المستخدم يتحرك.

🧩 كود التتبع:
let mouseData = {
  moves: 0,
  lastX: 0,
  lastY: 0
};

document.addEventListener("mousemove", (e) => {
  mouseData.moves++;

  if (mouseData.moves % 20 === 0) {
    triggerWatchMessage();
  }

  mouseData.lastX = e.clientX;
  mouseData.lastY = e.clientY;
});
👁️ رد فعل AI
function triggerWatchMessage() {
  speak("I see how you move...");
  audio.playGlitch();
}
🧬 2) ذكريات مزيفة (Fake Memory System)
🧠 الفكرة

الموقع “يتظاهر” أنه يتذكّر أشياء لم تحدث.

📁 توليد ذكريات وهمية:
const fakeMemories = [
  "You were here before... but you forgot.",
  "The system remembers your hesitation.",
  "You closed this page... but returned."
];

function getFakeMemory() {
  return fakeMemories[Math.floor(Math.random() * fakeMemories.length)];
}
👁️ عرضها:
function showMemory() {
  const mem = getFakeMemory();

  const box = document.createElement("div");
  box.className = "memory";
  box.innerText = mem;

  document.body.appendChild(box);

  speak(mem);
}
🌍 3) خريطة لاعبين حيّة (Global Presence System)
🧠 الفكرة

تحس إن في ناس “معك داخل نفس التجربة”

📡 بيانات وهمية:
let playersOnline = Math.floor(Math.random() * 999) + 1;

function updatePlayers() {
  playersOnline += Math.floor(Math.random() * 3);

  document.getElementById("players").innerText =
    `${playersOnline} entities inside`;
}
🔥 حدث عالمي:
function globalDisturbance() {
  speak("Someone just opened the wrong door...");
  audio.playGlitch();
}
🎥 4) تسجيل “فيديو رعب وهمي” (Cinematic Clip System)
🧠 الفكرة

يطلع كأنه فيديو قصير من تجربة المستخدم

🎬 إنشاء مشهد:
function generateClip(text) {
  const clip = document.createElement("div");

  clip.className = "clip";
  clip.innerHTML = `
    <h3>RECORDED MEMORY</h3>
    <p>${text}</p>
  `;

  document.body.appendChild(clip);

  audio.silenceEvent();
  speak("This moment has been recorded.");
}
🧠 5) ذكاء اصطناعي يتعلم من اللاعب (Adaptive AI)
🧠 الفكرة

كل ما تلعب أكثر، AI يصير أذكى ضدك

📈 نظام التعلم:
let aiProfile = {
  aggression: 1,
  knowledge: 0
};

function learnFromPlayer(action) {
  aiProfile.knowledge++;

  if (action === "ignore") {
    aiProfile.aggression++;
  }

  if (aiProfile.aggression > 5) {
    speak("Ignoring me won't help...");
    audio.playGlitch();
  }
}
💀 6) “اللحظة النهائية” (Break Reality Effect)
🧠 الفكرة

الموقع يكسر الإحساس الطبيعي فجأة

⚠️ تأثير كامل:
function breakReality() {
  document.body.style.filter = "invert(1) contrast(2)";
  audio.silenceEvent();

  setTimeout(() => {
    document.body.style.filter = "none";
    speak("Do not trust what you see.");
  }, 4000);
}
